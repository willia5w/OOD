5   import java.util.*;
6   
7   /**
8    * Iterator for ArrayList of Runners.
9    */
10  public class RunnersIterator implements Iterator {
11    private Stack <Runner> runnerStack;
12  
13    /**
14     * Iterates through runners in the list, qualifying throughout.
15     * @param runnerList A list of runners
16     */
17    public RunnersIterator(ArrayList<Runner> runnerList) {
18      for (int i = 0; i < runnerList.size(); i++) {
19        Runner currentRunner = runnerList.get(i);
20        qualified(currentRunner);
21      }
22    }
23  
24    /**
25     * Get the current stack of Runners.
26     * @return  current runner stack
27     */
28    public Stack <Runner> getRunnerStack() {
29      return runnerStack;
30    }
31  
32    /**
33     * Pushes runners to the stack if they are sufficiently experienced with 5K runs.
34     * Must also have half marathon race time under 100.00 minutes.
35     * @param runner A runner from the list of runners
36     */
37    public void qualified(Runner runner) {
38      int fastestHalfMarathon = runner.getHalfMarathonTimes().indexOf(Collections.min(runner.getHalfMarathonTimes()));
39      if (runner.getFiveKRaceTimes().size() > 4 && fastestHalfMarathon < 100) {
40        runnerStack.push(runner);
41      }
42    }
43  
44    /**
45     * Returns {@code true} if the iteration has more elements.
46     * (In other words, returns {@code true} if {@link #next} would
47     * return an element rather than throwing an exception.)
48     *
49     * @return {@code true} if the iteration has more elements
50     */
51    @Override
52    public boolean hasNext() {
53      return !runnerStack.isEmpty();
54    }
55  
56    /**
57     * Returns the next element in the iteration.
58     *
59     * @return the next element in the iteration
60     * @throws NoSuchElementException if the iteration has no more elements
61     */
62    @Override
63    public Object next() {
64      if (!hasNext()) {
65        throw new NoSuchElementException("No more qualified runners");
66      }
67  
68      Runner nextRunner = runnerStack.pop();
69      qualified(nextRunner);
70      return nextRunner.getName();
71    }
72  }
