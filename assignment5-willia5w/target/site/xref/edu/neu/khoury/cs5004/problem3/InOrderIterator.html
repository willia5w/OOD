3   import java.util.Iterator;
4   import java.util.NoSuchElementException;
5   import java.util.Stack;
6   
7   /**
8    * InOrderIterator for an IBT.
9    * Inspired by: http://n00tc0d3r.blogspot.com/2013/09/inorder-binary-tree-traversal-with.html?q=binary+tree
10   */
11  public class InOrderIterator implements Iterator {
12    Stack <Node> stack = new Stack<>();
13  
14  
15    /**
16     * Start pushing left nodes onto stack and then go to right nodes.
17     * @param cur current Node
18     */
19    private void pushLeftChildren(Node cur) {
20      while (cur != null) {
21        stack.push(cur);
22        cur = cur.getLeft();
23      }
24    }
25  
26    /**
27     * Constructor for the InOrderIterator.
28     */
29    public InOrderIterator(Node root) {
30      pushLeftChildren(root);
31    }
32  
33    @Override
34    public boolean hasNext() {
35      return !stack.isEmpty();
36    }
37  
38    @Override
39    public Integer next() {
40      if (!hasNext()) {
41        throw new NoSuchElementException("All nodes visited");
42      }
43  
44      Node res = stack.pop();
45      pushLeftChildren(res.getRight()); // After left, right
46  
47      return res.getValue();
48    }
49  }
